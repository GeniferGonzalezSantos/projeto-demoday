class Pessoa(models.Model):
    GENEROS = (
        ('F', 'Feminino'),
        ('O', 'Outro'),
    )
    nome = models.CharField(
        max_length=255,
        verbose_name='Nome'
    )
    sobrenome = models.CharField(
        max_length=255,
        verbose_name='Sobrenome'
    )

    genero = models.CharField(
        max_length=255,
        verbose_name='Gênero',
        choices=GENEROS
    )
    email = models.EmailField(
        max_length=255,
        verbose_name='E-mail'
    )
    
    idade = models.CharField(
        max_length=2,
        verbose_name= 'Idade'
    )
